-- 移网出账用户数
SELECT COUNT(DISTINCT USER_ID) 
FROM DWA_V_M_CUS_CB_USER_INFO 
WHERE MONTH_ID = '202603'
  AND SUBSTR(SERVICE_TYPE,1,2) IN ('40','50')
  AND IS_IOT = '0'
  AND IS_STAT = '1'
  AND IS_ACCT = '1';

-- 移网新发展用户数
SELECT COUNT(DISTINCT USER_ID) 
FROM DWA_V_M_CUS_CB_USER_INFO 
WHERE MONTH_ID = '202603'
  AND SUBSTR(SERVICE_TYPE,1,2) IN ('40','50')
  AND IS_IOT = '0'
  AND IS_STAT = '1'
  AND IS_THIS_DEV = '1';

-- 移网在网用户数
SELECT COUNT(DISTINCT USER_ID) 
FROM DWA_V_M_CUS_CB_USER_INFO 
WHERE MONTH_ID = '202603'
  AND SUBSTR(SERVICE_TYPE,1,2) IN ('40','50')
  AND IS_IOT = '0'
  AND IS_STAT = '1'
  AND IS_INNET = '1';
