-- 主查询：按省分、融合类型、是否行短、是否隐私号、是否物联网分组统计
SELECT 
    A.MONTH_ID,
    P.PROV_DESC AS PROV_NAME,
    CASE WHEN B.COMP_ID IS NULL THEN '单用户' ELSE '融合用户' END AS FUSION_TYPE,
    CASE WHEN A.NET_TYPE_ID = '65' AND A.BRAND_ID_CBSS IN ('HCA1', 'HDA1') THEN '是' ELSE '否' END AS IS_ROW_SMS,
    CASE WHEN C.PRODUCT_ID IS NOT NULL THEN '是' ELSE '否' END AS IS_PRIVACY_NUM,
    CASE WHEN A.IS_IOT = '1' OR D.PRODUCT_ID IS NOT NULL OR SUBSTR(A.DEVICE_NUMBER, 1, 4) IN ('1455', '1454', '1457', '1456') OR SUBSTR(A.DEVICE_NUMBER, 1, 3) IN ('146') THEN '是' ELSE '否' END AS IS_IOT_FLAG,
    COUNT(DISTINCT CASE WHEN A.IS_INNET = '1' THEN A.USER_ID END) AS INNET_USER_CNT,
    COUNT(DISTINCT CASE WHEN A.IS_ACCT = '1' THEN A.USER_ID END) AS ACCT_USER_CNT,
    COUNT(DISTINCT CASE WHEN A.IS_THIS_DEV = '1' THEN A.USER_ID END) AS NEW_DEV_USER_CNT
FROM DWA_V_M_CUS_CB_USER_INFO A
LEFT JOIN DWA_V_M_CUS_CB_OM_DATUM B ON A.USER_ID = B.USER_ID AND B.USER_RN_TYPE = '1' AND B.IS_USER_VALID = '1' AND B.IS_COMP_VALID = '1'
LEFT JOIN DIM_PRODUCT_PRIVACY_NUMBER C ON A.PRODUCT_ID = C.PRODUCT_ID
LEFT JOIN DIM_PRODUCT_WLW D ON A.PRODUCT_ID = D.PRODUCT_ID
LEFT JOIN DIM_PROV P ON A.PROV_ID = P.PROV_ID
WHERE A.MONTH_ID = '202601'
  AND SUBSTR(A.SERVICE_TYPE,1,2) IN ('40','50')
  AND A.IS_STAT = '1'
GROUP BY 
    A.MONTH_ID,
    P.PROV_DESC,
    CASE WHEN B.COMP_ID IS NULL THEN '单用户' ELSE '融合用户' END,
    CASE WHEN A.NET_TYPE_ID = '65' AND A.BRAND_ID_CBSS IN ('HCA1', 'HDA1') THEN '是' ELSE '否' END,
    CASE WHEN C.PRODUCT_ID IS NOT NULL THEN '是' ELSE '否' END,
    CASE WHEN A.IS_IOT = '1' OR D.PRODUCT_ID IS NOT NULL OR SUBSTR(A.DEVICE_NUMBER, 1, 4) IN ('1455', '1454', '1457', '1456') OR SUBSTR(A.DEVICE_NUMBER, 1, 3) IN ('146') THEN '是' ELSE '否' END;
